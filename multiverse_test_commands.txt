// G1axWorldGenerator - Multiverse Test Commands
// Copy and paste these commands in-game to create optimized worlds with different biomes
// All worlds automatically set to: Hard difficulty, No mob spawning, No tile drops, 1000 block border

// Original Biomes
/mv create plains_world normal -g G1axWorldGenerator:PLAINS
/mv create desert_world normal -g G1axWorldGenerator:DESERT
/mv create badlands_world normal -g G1axWorldGenerator:BADLANDS
/mv create mushroom_world normal -g G1axWorldGenerator:MUSHROOM

// New End Theme Biomes
/mv create end_world normal -g G1axWorldGenerator:END

// New Different Theme Biomes
/mv create nether_world normal -g G1axWorldGenerator:NETHER
/mv create snow_world normal -g G1axWorldGenerator:SNOW
/mv create ice_world normal -g G1axWorldGenerator:ICE
/mv create void_world normal -g G1axWorldGenerator:VOID

// Teleport Commands (use after creating worlds)
/mv tp plains_world
/mv tp desert_world
/mv tp badlands_world
/mv tp mushroom_world
/mv tp end_world
/mv tp nether_world
/mv tp snow_world
/mv tp ice_world
/mv tp void_world

// List all worlds
/mv list

// Remove worlds (if needed)
/mv remove plains_world
/mv remove desert_world
/mv remove badlands_world
/mv remove mushroom_world
/mv remove end_world
/mv remove nether_world
/mv remove snow_world
/mv remove ice_world
/mv remove void_world

// World Info Commands
/mv info plains_world
/mv info end_world
/mv info snow_world

// Set world spawn
/mv setspawn

// Available Biome Types:
// PLAINS - Grass terrain with trees and flowers
// DESERT - Sand terrain with cacti
// BADLANDS - Mesa/terracotta terrain
// MUSHROOM - Mushroom island terrain
// END - End stone terrain with chorus plants
// NETHER - Netherrack terrain with crimson trees and nether wart
// SNOW - Grass terrain with snow layers and spruce trees
// ICE - Frozen terrain with sharp ice spikes
// VOID - Floating islands in void space

// - Mob Spawning: Completely disabled
// - Auto Save: Disabled

// - Simulation Distance: 4 chunks

// Game Rules Optimized:
// - Day/Night Cycle: Disabled (always day)

// - Mob Griefing: Disabled
// - Random Tick Speed: 0 (no crop growth/decay)
// - Tile Drops: Disabled
// - Entity Drops: Disabled
// - Keep Inventory: Disabled
// - Advancement Announcements: Disabled

// Spawn Rate Optimizations:
// - Animal Spawns: 0 ticks (disabled)
